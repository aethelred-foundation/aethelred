Ship your first inference in 5 minutes

Multi-language SDKs with consistent API shape, async support, and production-ready docs.

submit_inference 
# pip install aethelred-sdk
import hashlib

from aethelred import AethelredClient, ProofType

client = AethelredClient("https://rpc.testnet.aethelred.org")
prompt = b'{"prompt":"Hello AI"}'

submit = client.jobs.submit(
    model_hash=bytes.fromhex("09f7e2b3c5d6a1f9..."),
    input_hash=hashlib.sha256(prompt).digest(),
    proof_type=ProofType.HYBRID,
    priority=5,
    max_gas=1_000_000,
    timeout_blocks=120,
)

job = client.jobs.wait_for_completion(submit.job_id)
seal = client.seals.verify(job.seal_id)
print(f"seal valid: {seal.valid}")
// npm install @aethelred/sdk
import { AethelredClient, Network, ProofType } from '@aethelred/sdk';

const client = new AethelredClient({ network: Network.TESTNET });

const submit = await client.jobs.submit({
  modelHash: '09f7e2b3c5d6a1f9...',
  inputHash: await sha256Hex('{"prompt":"Hello AI"}'),
  proofType: ProofType.HYBRID,
  priority: 5,
  maxGas: '1000000',
});

const job = await client.jobs.waitForCompletion(submit.jobId);
const seal = await client.seals.verify(job.sealId);
console.log(`seal valid: ${seal.valid}`);
// Cargo.toml: aethelred-sdk = "1"
use aethelred_sdk::{AethelredClient, Network, ProofType};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = AethelredClient::new(Network::Testnet).await?;

    let submit = client.jobs().submit(SubmitJobRequest {
        model_hash: "09f7e2b3c5d6a1f9...".into(),
        proof_type: Some(ProofType::Hybrid),
        priority: Some(5),
        max_gas: Some("1000000".into()),
        ..Default::default()
    }).await?;

    let job = client.jobs().wait_for_completion(&submit.job_id).await?;
    let seal = client.seals().verify(&job.seal_id).await?;
    println!("seal valid: {}", seal.valid);
    Ok(())
}
// go get github.com/aethelred/sdk-go
package main

import (
    "context"
    "fmt"
    "github.com/aethelred/sdk-go/client"
)

func main() {
    c, _ := client.NewClient(client.Testnet)
    submit, _ := c.Jobs.Submit(context.Background(), jobs.SubmitRequest{
        ModelHash: "09f7e2b3c5d6a1f9...",
        ProofType: types.ProofTypeHybrid,
        Priority:  5,
        MaxGas:    "1000000",
    })

    job, _ := c.Jobs.WaitForCompletion(ctx, submit.JobID)
    seal, _ := c.Seals.Verify(ctx, job.SealID)
    fmt.Printf("seal valid: %t\n", seal.Valid)
}

Quick Start

Get started in 3 steps

From zero to your first on-chain AI inference in under five minutes.

STEP 01

Install the CLI

Install the Aethelred CLI via Homebrew or Cargo on macOS, Linux, and WSL.

terminal 
brew install aethelred-foundation/tap/aeth
# or: cargo install aeth
STEP 02

Connect to Testnet

Configure the network and create a wallet. Fund it from the testnet faucet at launch.

terminal 
aeth config set --network testnet
aeth config set --rpc https://rpc.testnet.aethelred.io
aeth wallet create --name mykey
STEP 03

Submit your first job

Submit an AI inference job in one command while the network handles verification and settlement.

terminal 
aeth pouw submit-job \
  --model-hash abc123 \
  --input ./prompt.json \
  --verification-type hybrid \
  --from mykey
