// SPDX-License-Identifier: MIT pragma solidity ^0.8.19; import "@aethelred/contracts/ISealVerifier.sol"; /// @title AI Computation Verifier /// @notice Verify sealed AI computations on-chain contract AIVerifier { ISealVerifier public verifier; mapping(bytes32 => bool) public verifiedSeals; mapping(bytes32 => address) public sealOwners; event ComputationVerified( bytes32 indexed sealId, address indexed verifier, uint256 timestamp ); constructor(address _verifier) { verifier = ISealVerifier(_verifier); } function verifySeal( bytes32 sealId, bytes calldata proof ) external returns (bool) { require(!verifiedSeals[sealId], "Already verified"); require( verifier.verify(sealId, proof), "Invalid seal proof" ); verifiedSeals[sealId] = true; sealOwners[sealId] = msg.sender; emit ComputationVerified( sealId, msg.sender, block.timestamp ); return true; } }
