AIVerifier.sol
// SPDX-License-Identifier: MIT pragma solidity ^0.8.19; import "@aethelred/contracts/ISealVerifier.sol"; contract AIVerifier { ISealVerifier public verifier; mapping(bytes32 => bool) public verifiedSeals; event ComputationVerified( bytes32 sealId, address model, uint256 timestamp ); function verifySeal( bytes32 sealId, bytes calldata proof ) external returns (bool) { require( verifier.verify(sealId, proof), "Invalid seal" ); verifiedSeals[sealId] = true; emit ComputationVerified( sealId, msg.sender, block.timestamp ); return true; } }
